<h2 id="step-1">Step 1: Clone and onboard</h2>

<CommandTabs
  macos={`git clone <your-repo-url> BIOMETRICS\ncd BIOMETRICS\n./biometrics-onboard`}
  linux={`git clone <your-repo-url> BIOMETRICS\ncd BIOMETRICS\n./biometrics-onboard`}
  windowsWSL={`git clone <your-repo-url> BIOMETRICS\ncd BIOMETRICS\n./biometrics-onboard`}
/>

<VerifyBlock title="Verify onboarding output" expected="ready=true" testID="verify-onboarding">
  Run <code>cat .biometrics/onboard/report.json</code> and confirm no blocking failures.
</VerifyBlock>

<h2 id="step-2">Step 2: Start control plane</h2>

<CommandTabs
  macos={`cd biometrics-cli\ngo run ./cmd/controlplane`}
  linux={`cd biometrics-cli\ngo run ./cmd/controlplane`}
  windowsWSL={`cd biometrics-cli\ngo run ./cmd/controlplane`}
/>

<VerifyBlock title="Verify health" expected='{"status":"ok"}' testID="verify-health">
  In another terminal, run <code>curl -fsS http://127.0.0.1:59013/health</code>.
</VerifyBlock>

<h2 id="step-3">Step 3: Start first run</h2>

<CommandTabs
  macos={`curl -fsS -X POST http://127.0.0.1:59013/api/v1/runs \\\n  -H "Content-Type: application/json" \\\n  -d '{"project_id":"biometrics","goal":"quickstart run","mode":"autonomous"}'`}
  linux={`curl -fsS -X POST http://127.0.0.1:59013/api/v1/runs \\\n  -H "Content-Type: application/json" \\\n  -d '{"project_id":"biometrics","goal":"quickstart run","mode":"autonomous"}'`}
  windowsWSL={`curl -fsS -X POST http://127.0.0.1:59013/api/v1/runs \\\n  -H "Content-Type: application/json" \\\n  -d '{"project_id":"biometrics","goal":"quickstart run","mode":"autonomous"}'`}
/>

<VerifyBlock title="Verify run id" expected='"id":"run-..."' testID="verify-run-id">
  Response must include a non-empty run ID and an initial status.
</VerifyBlock>

<div className="ops-note">
